62// send value of type T1. Test received value of type T2 against expected value
63template <typename T1, typename T2>
64void testPrimitive(std::string label, T1 value, T2 expected) {
65 std::pair<std::string, T1> p("request_" + label, value);
66 MLRunner->populateFeatures(p);
67 T2 out = MLRunner->evaluate<T2>();
68 debug_out << " " << label << " reply: " << out << "\n";
69 if (std::abs(out - expected) > 10e-6) {
70 std::cerr << "Error: Expected " << label << " reply: " << expected
71 << ", Received: " << out << "\n";
72 exit(1);
73 }
74}

76template <typename T1, typename T2>
77void testVector(std::string label, std::vector<T1> value,
78 std::vector<T2> expected) {
79 std::pair<std::string, std::vector<T1>> p("request_" + label, value);
80 MLRunner->populateFeatures(p);
81 T2 *out;
82 size_t size;
83 MLRunner->evaluate<T2 *>(out, size);
84 std::vector<T2> reply(out, out + size);
85 debug_out << " " << label << " reply: ";
86 int i = 0;
87 for (auto x : reply) {
88 debug_out << x << " ";
89 if (std::abs(x - expected[i]) > 10e-6) {
90 std::cerr << "Error: Expected " << label << " reply: " << expected[i]
91 << ", Received: " << x << "\n";
92 exit(1);
93 }
94 i++;
95 }
96 debug_out << "\n";
97}

135int testPipes() {
136 if (pipe_name == "") {
137 std::cerr
138 << "Pipe name must be specified via --test-pipe-name=<filename>\n";
139 exit(1);
140 }
141 basename = "/tmp/" + pipe_name;
142 if (data_format == "json")
143 SerDesType = BaseSerDes::Kind::Json;
144 else if (data_format == "protobuf")
145 SerDesType = BaseSerDes::Kind::Protobuf;
146 else if (data_format == "bytes")
147 SerDesType = BaseSerDes::Kind::Bitstream;
148 else {
149 std::cout << "Invalid data format\n";
150 exit(1);
151 }
152
153 MLRunner = std::make_unique<PipeModelRunner>(
154 basename + ".out", basename + ".in", SerDesType, nullptr);
155
156 runTests();
157 return 0;
158}

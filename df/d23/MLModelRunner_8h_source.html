11// The MLModelRunner class is the main interface for interacting with the
12// ML models. The MLCompilerBridge uses the MLModelRunner class to set the
13// features to be sent to the model and get the result back from the model.
14//
15// This class internally uses the SerDes class to serialize and deserialize the
16// features and result.
17//
18// The MLModelRunner class is an abstract class and cannot be instantiated.
19//
20// This class internally uses the SerDes class to serialize and deserialize the
21// features and result.
22//
23// Supporting new Model Runners:
24// 1. Create a new class inheriting the MLModelRunner class.
25// 2. Override evaluateUntyped() method to call the model and get the result.
26//
27// Using any of the existing Model Runners:
28// 1. Instantiate the model runner object with the appropriate arguments.
29// 2. Call populateFeatures() to set the features to be sent to the model.
30// 3. Call evaluate() to get the send and receive the result from the model.
31// Similar flows apply for both training and inference.
32
33//===----------------------------------------------------------------------===//

65 template <typename T>
66 typename std::enable_if<
67 std::is_fundamental<typename std::remove_pointer<T>::type>::value,
68 void>::type
69 evaluate(T &data, size_t &dataSize) {
70 using BaseType = typename std::remove_pointer<T>::type;
71 void *res = evaluateUntyped();
72 T ret = static_cast<T>(malloc(SerDes->getMessageLength()));
73 memcpy(ret, res, SerDes->getMessageLength());
74 dataSize = SerDes->getMessageLength() / sizeof(BaseType);
75 data = ret;
76 }

28void ONNXModel::run(llvm::SmallVector<float, 100> &input,
29 llvm::SmallVector<float, 100> &output) {
30 Ort::AllocatorWithDefaultOptions allocator;
31 auto inputName = session->GetInputNameAllocated(0, allocator);
32 auto inputNameStr = inputName.get();
33
34 auto typeInfo = session->GetInputTypeInfo(0);
35 auto tensorInfo = typeInfo.GetTensorTypeAndShapeInfo();
36 auto inputDims = tensorInfo.GetShape();
37 std::replace_if(
38 inputDims.begin(), inputDims.end(), [](int64_t &i) { return i < 0; }, 1);
39
40 size_t inputTensorSize = std::accumulate(inputDims.begin(), inputDims.end(),
41 1, std::multiplies<int>());
42 assert(inputTensorSize == input.size());
43
44 auto memory_info =
45 Ort::MemoryInfo::CreateCpu(OrtArenaAllocator, OrtMemTypeDefault);
46 auto inputTmp = Ort::Value::CreateTensor<float>(
47 memory_info, input.data(), inputTensorSize, inputDims.data(),
48 inputDims.size());
49 auto inputTensor = &inputTmp;
50 assert(inputTensor->IsTensor());
51
52 auto outputName = session->GetOutputNameAllocated(0, allocator);
53 auto outputNameStr = outputName.get();
54 // std::cout << inputNameStr << " --- " << outputNameStr << "\n";
55 auto outputTensors = session->Run(Ort::RunOptions{nullptr}, &inputNameStr,
56 inputTensor, 1, &outputNameStr, 1);
57 assert(outputTensors.size() == 1 && outputTensors.front().IsTensor());
58
59 auto outputDims =
60 outputTensors.front().GetTensorTypeAndShapeInfo().GetShape()[1];
61
62 auto outVal = outputTensors.front().GetTensorMutableData<float>();
63
64 output = llvm::SmallVector<float, 100>(outVal, outVal + outputDims);
65 std::replace_if(
66 output.begin(), output.end(), [](double x) { return std::isnan(x); },
67 -1.17549e+038);
68}

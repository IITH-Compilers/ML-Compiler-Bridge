28Ort::Value ONNXModel::getInputValue(llvm::SmallVector<float, 100> &input, int inputIdx) {
29 auto typeInfo = session->GetInputTypeInfo(inputIdx);
30 auto tensorInfo = typeInfo.GetTensorTypeAndShapeInfo();
31 auto inputDims = tensorInfo.GetShape();
32 std::replace_if(
33 inputDims.begin(), inputDims.end(), [](int64_t &i) { return i < 0; }, 1);
34
35 size_t inputTensorSize = std::accumulate(inputDims.begin(), inputDims.end(),
36 1, std::multiplies<int>());
37 assert(inputTensorSize == input.size());
38
39 auto memory_info =
40 Ort::MemoryInfo::CreateCpu(OrtArenaAllocator, OrtMemTypeDefault);
41 auto inputTmp = Ort::Value::CreateTensor<float>(
42 memory_info, input.data(), inputTensorSize, inputDims.data(),
43 inputDims.size());
44 auto inputTensor = &inputTmp;
45 assert(inputTensor->IsTensor());
46 return inputTmp;
47}

49void ONNXModel::run(llvm::SmallVector<float, 100> &input,
50 llvm::SmallVector<float, 100> &output) {
51 Ort::AllocatorWithDefaultOptions allocator;
52
53 int input_count = session->GetInputCount();
54 llvm::SmallVector<llvm::SmallVector<float, 100>, 10> inputList;
55 inputList.push_back(input);
56 llvm::SmallVector<float, 100> dummy_input;
57 dummy_input.insert(dummy_input.end(), 1);
58 for (int i = 1; i < input_count; i++) {
59 inputList.push_back(dummy_input);
60 }
61
62 llvm::SmallVector<std::string, 10> inputNameList;
63 for (int i = 0; i < input_count; i++) {
64 auto inputName = session->GetInputNameAllocated(i, allocator);
65 auto inputNameStr = inputName.get();
66 inputNameList.push_back(inputNameStr);
67 }
68
69 std::vector<Ort::Value> input_final;
70 llvm::SmallVector<const char *, 10> inputNameStr_final;
71
72 for (int i = 0; i < input_count; i++){
73 input_final.insert(input_final.end(), getInputValue(inputList[i], i));
74 inputNameStr_final.insert(inputNameStr_final.end(), inputNameList[i].c_str());
75 }
76
77 auto outputName = session->GetOutputNameAllocated(0, allocator);
78 auto outputNameStr = outputName.get();
79
80 auto outputTensors =
81 session->Run(Ort::RunOptions{nullptr}, inputNameStr_final.data(),
82 input_final.data(), input_count, &outputNameStr, 1);
83
84 assert(outputTensors.size() == 1 && outputTensors.front().IsTensor());
85
86 auto outputDims =
87 outputTensors.front().GetTensorTypeAndShapeInfo().GetShape()[1];
88
89 auto outVal = outputTensors.front().GetTensorMutableData<float>();
90
91 output = llvm::SmallVector<float, 100>(outVal, outVal + outputDims);
92 std::replace_if(
93 output.begin(), output.end(), [](double x) { return std::isnan(x); },
94 -1.17549e+038);
95}

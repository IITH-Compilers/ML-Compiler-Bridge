64// === String encodings ===
65//
66// JSON strings are character sequences (not byte sequences like std::string).
67// We need to know the encoding, and for simplicity only support UTF-8.
68//
69// - When parsing, invalid UTF-8 is a syntax error like any other
70//
71// - When creating Values from strings, callers must ensure they are UTF-8.
72// with asserts on, invalid UTF-8 will crash the program
73// with asserts off, we'll substitute the replacement character (U+FFFD)
74// Callers can use json::isUTF8() and json::fixUTF8() for validation.
75//
76// - When retrieving strings from Values (e.g. asString()), the result will
77// always be valid UTF-8.
78

80constexpr bool
81 is_uint_64_bit_v = std::is_integral_v<T> &&std::is_unsigned_v<T> &&
82 sizeof(T) == sizeof(uint64_t);

427 // Succeeds if the Value is a Number, and exactly representable as int64_t.
428 std::optional<int64_t> getAsInteger() const {
429 if (LLVM_LIKELY(Type == T_Integer))
430 return as<int64_t>();
431 if (LLVM_LIKELY(Type == T_UINT64)) {
432 uint64_t U = as<uint64_t>();
433 if (LLVM_LIKELY(U <= uint64_t(std::numeric_limits<int64_t>::max()))) {
434 return U;
435 }
436 }
437 if (LLVM_LIKELY(Type == T_Double)) {
438 double D = as<double>();
439 if (LLVM_LIKELY(std::modf(D, &D) == 0.0 &&
440 D >= double(std::numeric_limits<int64_t>::min()) &&
441 D <= double(std::numeric_limits<int64_t>::max())))
442 return D;
443 }
444 return std::nullopt;
445 }
446 std::optional<uint64_t> getAsUINT64() const {
447 if (Type == T_UINT64)
448 return as<uint64_t>();
449 else if (Type == T_Integer) {
450 int64_t N = as<int64_t>();
451 if (N >= 0)
452 return as<uint64_t>();
453 }
454 return std::nullopt;
455 }

591 if (C.Owned) {
592 Owned.reset(new std::string(*C.Owned));

594 } else {

596 }

598 }

773template <typename T>
774bool fromJSON(const Value &E, std::optional<T> &Out, Path P) {
775 if (E.getAsNull()) {
776 Out = std::nullopt;
777 return true;
778 }
779 T Result = {};
780 if (!fromJSON(E, Result, P))
781 return false;
782 Out = std::move(Result);
783 return true;
784}
785template <typename T>
786bool fromJSON(const Value &E, std::vector<T> &Out, Path P) {
787 if (auto *A = E.getAsArray()) {
788 Out.clear();
789 Out.resize(A->size());
790 for (size_t I = 0; I < A->size(); ++I)
791 if (!fromJSON((*A)[I], Out[I], P.index(I)))
792 return false;
793 return true;
794 }
795 P.report("expected array");
796 return false;
797}
798template <typename T>
799bool fromJSON(const Value &E, std::map<std::string, T> &Out, Path P) {
800 if (auto *O = E.getAsObject()) {
801 Out.clear();
802 for (const auto &KV : *O)
803 if (!fromJSON(KV.second, Out[std::string(llvm::StringRef(KV.first))],
804 P.field(KV.first)))
805 return false;
806 return true;
807 }
808 P.report("expected object");
809 return false;
810}

95ONNXModelRunner *createONNXModelRunner(Environment *env, int numAgents, ...) {
96 assert(env != nullptr && "Environment is null!");
97
98 va_list args;
99 va_start(args, numAgents);
100 std::map<std::string, Agent *> agents;
101
102 for (int i = 0; i < numAgents; i += 2) {
103 char *agentName = va_arg(args, char *);
104 char *agentPath = va_arg(args, char *);
105 agents[agentName] = new Agent(agentPath);
106 }
107
108 va_end(args);
109
110 ONNXModelRunner *obj = new ONNXModelRunner(env, std::move(agents));

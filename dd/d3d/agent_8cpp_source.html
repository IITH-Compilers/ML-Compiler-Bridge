14Agent::Agent(std::string modelPath) {
15 // Create model object here
16 this->model = new ONNXModel(modelPath.c_str());
17}

19unsigned Agent::computeAction(Observation &input) {
20 // Call model on input
21 assert(input.size() > 0);
22 llvm::SmallVector<float, 100> model_input(input.begin(), input.end());
23 llvm::SmallVector<float, 100> model_output;
24
25 this->model->run(model_input, model_output);
26
27 // select action from model output
28 auto max = std::max_element(model_output.begin(),
29 model_output.end()); // [2, 4)
30 int argmaxVal = std::distance(model_output.begin(), max);
31
32 LLVM_DEBUG(
33 llvm::errs() << "---------------MODEL OUTPUT VECTOR:----------------\n");
34 for (auto e : model_output) {
35 LLVM_DEBUG(llvm::errs() << e << " ");
36 }
37 LLVM_DEBUG(llvm::errs() << "\nmax value and index are " << *max << " "
38 << argmaxVal << "\n");
39 return argmaxVal;
40}

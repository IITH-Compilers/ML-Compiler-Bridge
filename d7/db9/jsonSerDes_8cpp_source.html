57void *JsonSerDes::desJson(json::Value *V) {
58 switch (V->kind()) {
59 case json::Value::Kind::Null:
60 return nullptr;
61 case json::Value::Kind::Number: {
62 if (auto x = V->getAsInteger()) {
63 IntegerType *ret = new IntegerType();
64 *ret = x.value();
65 this->MessageLength = sizeof(IntegerType);
66 return ret;
67 } else if (auto x = V->getAsNumber()) {
68 RealType *ret = new RealType();
69 *ret = x.value();
70 this->MessageLength = sizeof(RealType);
71 return ret;
72 } else {
73 std::cerr << "Error in desJson: Number is not int, or double\n";
74 exit(1);
75 }
76 }
77 case json::Value::Kind::String: {
78 std::string *ret = new std::string();
79 *ret = V->getAsString()->str();
80 this->MessageLength = ret->size() * sizeof(char);
81 return ret->data();
82 }
83 case json::Value::Kind::Boolean: {
84 bool *ret = new bool();
85 *ret = V->getAsBoolean().value();
86 this->MessageLength = sizeof(bool);
87 return ret;
88 }
89 case json::Value::Kind::Array: {
90 // iterate over array and find its type
91 // assume all elements are of same type and return vector of that type
92 // if not, return nullptr
93 auto arr = V->getAsArray();
94
95 auto it = arr->begin();
96 auto first = it;
97 switch (first->kind()) {
98 case json::Value::Kind::Number: {
99 if (auto x = first->getAsInteger()) {
100 std::vector<IntegerType> *ret = new std::vector<IntegerType>();
101 for (auto it : *arr) {
102 ret->push_back(it.getAsInteger().value());
103 }
104 this->MessageLength = ret->size() * sizeof(IntegerType);
105 return ret->data();
106 } else if (auto x = first->getAsNumber()) {
107 std::vector<RealType> *ret = new std::vector<RealType>();
108 for (auto it : *arr) {
109 ret->push_back(it.getAsNumber().value());
110 }
111 this->MessageLength = ret->size() * sizeof(RealType);
112 return ret->data();
113 } else {
114 std::cerr << "Error in desJson: Number is not int, or double\n";
115 exit(1);
116 }
117 }
118 case json::Value::Kind::String: {
119 std::vector<std::string> *ret = new std::vector<std::string>();
120 for (auto it : *arr) {
121 ret->push_back(it.getAsString()->str());
122 }
123 this->MessageLength = ret->size() * sizeof(std::string);
124 return ret->data();
125 }
126 case json::Value::Kind::Boolean: {
127 std::vector<uint8_t> *ret = new std::vector<uint8_t>();
128 for (auto it : *arr) {
129 ret->push_back(it.getAsBoolean().value());
130 }
131 this->MessageLength = ret->size() * sizeof(uint8_t);
132 return ret->data();
133 }
134 default: {
135 std::cerr << "Error in desJson: Array is not of supported type\n";
136 exit(1);
137 }
138 }
139 }
140 }
141 return nullptr;
142}

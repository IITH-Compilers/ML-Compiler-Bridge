117 bool isPortAvailable(std::string addr) {
118 int max_retries = 30, attempts = 0;
119 double wait_seconds = 0.2, backoff_exp = 1.2;
120
121 int idx = addr.find(":");
122 int port = stoi(addr.substr(idx + 1, addr.size() - idx - 1));
123
124 while (attempts < max_retries) {
125 std::string command = "lsof -i :" + std::to_string(port);
126 FILE *pipe = popen(command.c_str(), "r");
127 if (!pipe) {
128 std::cerr << "Error executing command: " << std::strerror(errno)
129 << std::endl;
130 return false;
131 }
132
133 char buffer[256];
134 std::string result = "";
135 while (!feof(pipe)) {
136 if (fgets(buffer, 256, pipe) != nullptr)
137 result += buffer;
138 }
139 pclose(pipe);
140
141 if (result.empty()) {
142 return true;
143 }
144 attempts++;
145 std::cout << "Port is unavailable retrying! attempt: " << attempts
146 << std::endl;
147 std::this_thread::sleep_for(std::chrono::duration<double>(wait_seconds));
148 wait_seconds *= backoff_exp;
149 }
150
151 std::cout << "Port is unavailable now!" << std::endl;
152 return false;

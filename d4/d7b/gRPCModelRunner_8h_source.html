9// gRPCModelRunner class supporting communication via gRPC. This class is used
10// to communicate with the gRPC server and send/receive data to/from the model.
11// Supports interleaved communication with the model.
12//
13// There are two ways to use this class:
14// 1. Training mode - gRPC Server: In this mode, the gRPCModelRunner object is
15// created with the server address and the service object. The service object is
16// used to create the server and the server waits for the client to connect.
17// Once the client connects, the server waits for the client to send the request
18// and then responds with the result.
19// In Training mode, GrpcCompilerInterface class from GrpcCompilerInterface.py
20// acts as the client and sends the request to the server.
21//
22// 2. Inference mode - gRPC Client: In this mode, the gRPCModelRunner object is
23// created with the server address, request and response objects. The request
24// object is used to send the features to the model and the response object is
25// used to receive the result from the model.
26//
27// In Inference mode, the compiler pass using this class acts as the client and
28// sends the request to the server implemented by using
29// GrpcCompilerInterface.py.
30//
31//===---------------------------------------------------------------------===//
32//
33// Interfacing with the model using protobuf (.proto) files:
34// Users should define the service, stub, request and response by writing a
35// .proto file. The service and stub objects are generated using the protoc
36// compiler. The request and response objects are generated using the protoc
37// compiler or created by the user. The build process will automatically compile
38// .proto files and generates the relevant stubs and request/response objects in
39// both CPP and Python which will be used by gRPCModelRunner and
40// GrpcCompilerInterface respectively..
41//
42// In Inference mode, users should override `getAdvice()` RPC method in the
43// Python model code to process the request and send the response back to the
44// client. This method is called by the gRPC server in the evaluate_untyped()
45// method of gRPCModelRunner class to get the result from the model after
46// populating the features in the request object.
47//
48// In Training mode, users should override the RPC function/service that they
49// declare in the .proto file in the Compiler pass which is using the
50// gRPCModelRunner. This function is called by the gRPC client (Python model)
51// that is using the GrpcCompilerInterface class to send the request to the
52// server.
53//
54// Usage:
55// 1. Create a .proto file with the service and message definitions
56// 2. Generate the stubs using protoc
57// 3. Create a gRPCModelRunner object with the server address, stub, request and
58// response
59// 4. Populate the features to be sent to the model
60// 5. Call evaluate() to get the result back from the model
61//
62// ===----------------------------------------------------------------------===//

114 // checks whether a port number is available or not
115 bool isPortAvailable(std::string addr) {
116 int max_retries = 30, attempts = 0;
117 double wait_seconds = 0.2, backoff_exp = 1.2;
118
119 int idx = addr.find(":");
120 int port = stoi(addr.substr(idx + 1, addr.size() - idx - 1));
121
122 while (attempts < max_retries) {
123 std::string command = "lsof -i :" + std::to_string(port);
124 FILE* pipe = popen(command.c_str(), "r");
125 if (!pipe) {
126 std::cerr << "Error executing command: " << std::strerror(errno) << std::endl;
127 return false;
128 }
129
130 char buffer[256];
131 std::string result = "";
132 while (!feof(pipe)) {
133 if (fgets(buffer, 256, pipe) != nullptr)
134 result += buffer;
135 }
136 pclose(pipe);
137
138 if (result.empty()) {
139 return true;
140 }
141 attempts++;
142 std::cout << "Port is unavailable retrying! attempt: " << attempts << std::endl;
143 std::this_thread::sleep_for(std::chrono::duration<double>(wait_seconds));
144 wait_seconds *= backoff_exp;
145 }
146
147 std::cout << "Port is unavailable now!" << std::endl;
148 return false;
149 }

154 void *evaluateUntyped() override {
155 assert(!server_mode &&
156 "evaluateUntyped not implemented for gRPCModelRunner; "
157 "Override gRPC method instead");
158 assert(request != nullptr && "Request cannot be null");
159
160 int max_retries = 30, attempts = 0;
161 double retries_wait_secs = 0.2;
162 int deadline_time = 1000;
163 int deadline_max_retries = 30, deadline_attpts = 0;
164 double retry_wait_backoff_exponent = 1.5;
165
166 // setting a deadline
167 auto deadline = std::chrono::system_clock::now() + std::chrono::milliseconds(deadline_time);
168
169 while (attempts < max_retries && deadline_attpts < deadline_max_retries) {
170 grpc::ClientContext grpcCtx;
171 request = getRequest();
172 grpc::Status status;
173 grpcCtx.set_deadline(deadline);
174
175 status = stub_->getAdvice(&grpcCtx, *request, response);
176
177 if (status.error_code() == grpc::StatusCode::DEADLINE_EXCEEDED) {
178 deadline_attpts++;
179 int ext_deadline = 2 * deadline_time;
180 deadline_time = ext_deadline;
181 std::cout << "Deadline Exceeded for Request! sending the message again with extended deadline : " << deadline_time << "\n";
182 deadline = std::chrono::system_clock::now() + std::chrono::milliseconds(deadline_time);
183 }
184 else if (status.error_code() == grpc::StatusCode::UNAVAILABLE) {
185 attempts++;
186 std::cout << "Server is unavailable retrying! attempt: " << attempts << "\n";
187 std::this_thread::sleep_for(std::chrono::duration<double>(retries_wait_secs));
188 retries_wait_secs *= retry_wait_backoff_exponent;
189 }
190 else {
191 request->Clear();
192 if (!status.ok())
193 if (Ctx)
194 Ctx->emitError("gRPC failed: " + status.error_message());
195 else
196 std::cerr << "gRPC failed: " << status.error_message() << std::endl;
197 // auto *action = new int(); // Hard wired for PosetRL case, should be fixed
198 // *action = response->action();
199 // return action;
200 return SerDes->deserializeUntyped(response);
201 }
202 }
203
204 std::cout << "Server is unavailable now!!!\n";
205 return new int(-1);

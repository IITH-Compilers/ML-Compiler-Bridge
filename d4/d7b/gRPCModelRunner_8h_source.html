113 bool isPortAvailable(std::string addr) {
114 int max_retries = 30, attempts = 0;
115 double wait_seconds = 0.2, backoff_exp = 1.2;
116
117 int idx = addr.find(":");
118 int port = stoi(addr.substr(idx + 1, addr.size() - idx - 1));
119
120 while (attempts < max_retries) {
121 std::string command = "lsof -i :" + std::to_string(port);
122 FILE *pipe = popen(command.c_str(), "r");
123 if (!pipe) {
124 std::cerr << "Error executing command: " << std::strerror(errno)
125 << std::endl;
126 return false;
127 }
128
129 char buffer[256];
130 std::string result = "";
131 while (!feof(pipe)) {
132 if (fgets(buffer, 256, pipe) != nullptr)
133 result += buffer;
134 }
135 pclose(pipe);
136
137 if (result.empty()) {
138 return true;
139 }
140 attempts++;
141 std::cout << "Port is unavailable retrying! attempt: " << attempts
142 << std::endl;
143 std::this_thread::sleep_for(std::chrono::duration<double>(wait_seconds));
144 wait_seconds *= backoff_exp;
145 }
146
147 std::cout << "Port is unavailable now!" << std::endl;
148 return false;

180 deadline_attpts++;
181 int ext_deadline = 2 * deadline_time;
182 deadline_time = ext_deadline;
183 std::cout << "Deadline Exceeded for Request! sending the message again "
184 "with extended deadline : "
185 << deadline_time << "\n";
186 deadline = std::chrono::system_clock::now() +
187 std::chrono::milliseconds(deadline_time);
188 } else if (status.error_code() == grpc::StatusCode::UNAVAILABLE) {
189 attempts++;
190 std::cout << "Server is unavailable retrying! attempt: " << attempts
191 << "\n";
192 std::this_thread::sleep_for(

221 int RunService(grpc::Service *s) {
222 exit_requested = new std::promise<void>();
223 grpc::ServerBuilder builder;
224 // if (!this->isPortAvailable(server_address)) return -1;
225 builder.AddListeningPort(server_address, grpc::InsecureServerCredentials());
226 builder.RegisterService(s);
227 std::unique_ptr<grpc::Server> server(builder.BuildAndStart());
228 std::cout << "Server Listening on " << server_address << std::endl;
229 auto serveFn = [&]() { server->Wait(); };
230 std::thread serving_thread(serveFn);
231 auto f = exit_requested->get_future();
232 this->requestExit();
233 f.wait();
234 server->Shutdown();
235 serving_thread.join();
236 std::cout << "Server Shutdowns Successfully" << std::endl;
237 return 0;
238 }

192bool operator==(const Value &L, const Value &R) {
193 if (L.kind() != R.kind())
194 return false;
195 switch (L.kind()) {
196 case Value::Null:
197 return *L.getAsNull() == *R.getAsNull();
198 case Value::Boolean:
199 return *L.getAsBoolean() == *R.getAsBoolean();
200 case Value::Number:
201 // Workaround for https://gcc.gnu.org/bugzilla/show_bug.cgi?id=323
202 // The same integer must convert to the same double, per the standard.
203 // However we see 64-vs-80-bit precision comparisons with gcc-7 -O3 -m32.
204 // So we avoid floating point promotion for exact comparisons.
205 if (L.Type == Value::T_Integer || R.Type == Value::T_Integer)
206 return L.getAsInteger() == R.getAsInteger();
207 return *L.getAsNumber() == *R.getAsNumber();
208 case Value::String:
209 return *L.getAsString() == *R.getAsString();
210 case Value::Array:
211 return *L.getAsArray() == *R.getAsArray();
212 case Value::Object:
213 return *L.getAsObject() == *R.getAsObject();
214 }
215 llvm_unreachable("Unknown value kind");
216}

218void Path::report(llvm::StringLiteral Msg) {
219 // Walk up to the root context, and count the number of segments.
220 unsigned Count = 0;
221 const Path *P;
222 for (P = this; P->Parent != nullptr; P = P->Parent)
223 ++Count;
224 Path::Root *R = P->Seg.root();
225 // Fill in the error message and copy the path (in reverse order).
226 R->ErrorMessage = Msg;
227 R->ErrorPath.resize(Count);
228 auto It = R->ErrorPath.begin();
229 for (P = this; P->Parent != nullptr; P = P->Parent)
230 *It++ = P->Seg;
231}

233Error Path::Root::getError() const {
234 std::string S;
235 raw_string_ostream OS(S);
236 OS << (ErrorMessage.empty() ? "invalid JSON contents" : ErrorMessage);
237 if (ErrorPath.empty()) {
238 if (!Name.empty())
239 OS << " when parsing " << Name;
240 } else {
241 OS << " at " << (Name.empty() ? "(root)" : Name);
242 for (const Path::Segment &S : llvm::reverse(ErrorPath)) {
243 if (S.isField())
244 OS << '.' << S.field();
245 else
246 OS << '[' << S.index() << ']';
247 }
248 }
249 return createStringError(llvm::inconvertibleErrorCode(), OS.str());

265// Prints a one-line version of a value that isn't our main focus.
266// We interleave writes to OS and JOS, exploiting the lack of extra buffering.
267// This is OK as we own the implementation.
268void abbreviate(const Value &V, OStream &JOS) {
269 switch (V.kind()) {
270 case Value::Array:
271 JOS.rawValue(V.getAsArray()->empty() ? "[]" : "[ ... ]");
272 break;
273 case Value::Object:
274 JOS.rawValue(V.getAsObject()->empty() ? "{}" : "{ ... }");
275 break;
276 case Value::String: {
277 llvm::StringRef S = *V.getAsString();
278 if (S.size() < 40) {
279 JOS.value(V);
280 } else {
281 std::string Truncated = fixUTF8(S.take_front(37));
282 Truncated.append("...");
283 JOS.value(Truncated);
284 }
285 break;
286 }
287 default:
288 JOS.value(V);
289 }
290}

292// Prints a semi-expanded version of a value that is our main focus.
293// Array/Object entries are printed, but not recursively as they may be huge.
294void abbreviateChildren(const Value &V, OStream &JOS) {
295 switch (V.kind()) {
296 case Value::Array:
297 JOS.array([&] {
298 for (const auto &I : *V.getAsArray())
299 abbreviate(I, JOS);
300 });
301 break;
302 case Value::Object:
303 JOS.object([&] {
304 for (const auto *KV : sortedElements(*V.getAsObject())) {
305 JOS.attributeBegin(KV->first);
306 abbreviate(KV->second, JOS);
307 JOS.attributeEnd();
308 }
309 });
310 break;
311 default:
312 JOS.value(V);
313 }
314}

318void Path::Root::printErrorContext(const Value &R, raw_ostream &OS) const {
319 OStream JOS(OS, /*IndentSize=*/2);
320 // PrintValue recurses down the path, printing the ancestors of our target.
321 // Siblings of nodes along the path are printed with abbreviate(), and the
322 // target itself is printed with the somewhat richer abbreviateChildren().
323 // 'Recurse' is the lambda itself, to allow recursive calls.
324 auto PrintValue = [&](const Value &V, ArrayRef<Segment> Path, auto &Recurse) {
325 // Print the target node itself, with the error as a comment.
326 // Also used if we can't follow our path, e.g. it names a field that
327 // *should* exist but doesn't.
328 auto HighlightCurrent = [&] {
329 std::string Comment = "error: ";
330 Comment.append(ErrorMessage.data(), ErrorMessage.size());
331 JOS.comment(Comment);
332 abbreviateChildren(V, JOS);
333 };
334 if (Path.empty()) // We reached our target.
335 return HighlightCurrent();
336 const Segment &S = Path.back(); // Path is in reverse order.
337 if (S.isField()) {
338 // Current node is an object, path names a field.
339 llvm::StringRef FieldName = S.field();
340 const Object *O = V.getAsObject();
341 if (!O || !O->get(FieldName))
342 return HighlightCurrent();
343 JOS.object([&] {
344 for (const auto *KV : sortedElements(*O)) {
345 JOS.attributeBegin(KV->first);
346 if (FieldName.equals(KV->first))
347 Recurse(KV->second, Path.drop_back(), Recurse);
348 else
349 abbreviate(KV->second, JOS);
350 JOS.attributeEnd();
351 }
352 });
353 } else {
354 // Current node is an array, path names an element.
355 const Array *A = V.getAsArray();
356 if (!A || S.index() >= A->size())
357 return HighlightCurrent();
358 JOS.array([&] {
359 unsigned Current = 0;
360 for (const auto &V : *A) {
361 if (Current++ == S.index())
362 Recurse(V, Path.drop_back(), Recurse);
363 else
364 abbreviate(V, JOS);
365 }
366 });
367 }
368 };
369 PrintValue(R, ErrorPath, PrintValue);
370}

380 size_t ErrOffset;

382 return true;

425bool Parser::parseValue(Value &Out) {
426 eatWhitespace();
427 if (P == End)
428 return parseError("Unexpected EOF");
429 switch (char C = next()) {
430 // Bare null/true/false are easy - first char identifies them.
431 case 'n':
432 Out = nullptr;
433 return (next() == 'u' && next() == 'l' && next() == 'l') ||
434 parseError("Invalid JSON value (null?)");
435 case 't':
436 Out = true;
437 return (next() == 'r' && next() == 'u' && next() == 'e') ||
438 parseError("Invalid JSON value (true?)");
439 case 'f':
440 Out = false;
441 return (next() == 'a' && next() == 'l' && next() == 's' && next() == 'e') ||
442 parseError("Invalid JSON value (false?)");
443 case '"': {
444 std::string S;
445 if (parseString(S)) {
446 Out = std::move(S);
447 return true;
448 }
449 return false;
450 }
451 case '[': {
452 Out = Array{};
453 Array &A = *Out.getAsArray();
454 eatWhitespace();
455 if (peek() == ']') {
456 ++P;
457 return true;
458 }
459 for (;;) {
460 A.emplace_back(nullptr);
461 if (!parseValue(A.back()))
462 return false;
463 eatWhitespace();
464 switch (next()) {
465 case ',':
466 eatWhitespace();
467 continue;
468 case ']':
469 return true;
470 default:
471 return parseError("Expected , or ] after array element");
472 }
473 }
474 }
475 case '{': {
476 Out = Object{};
477 Object &O = *Out.getAsObject();
478 eatWhitespace();
479 if (peek() == '}') {
480 ++P;
481 return true;
482 }
483 for (;;) {
484 if (next() != '"')
485 return parseError("Expected object key");
486 std::string K;
487 if (!parseString(K))
488 return false;
489 eatWhitespace();
490 if (next() != ':')
491 return parseError("Expected : after object key");
492 eatWhitespace();
493 if (!parseValue(O[std::move(K)]))
494 return false;
495 eatWhitespace();
496 switch (next()) {
497 case ',':
498 eatWhitespace();
499 continue;
500 case '}':
501 return true;
502 default:
503 return parseError("Expected , or } after object property");
504 }
505 }
506 }
507 default:
508 if (isNumber(C))
509 return parseNumber(C, Out);
510 return parseError("Invalid JSON value");
511 }
512}

514bool Parser::parseNumber(char First, Value &Out) {
515 // Read the number into a string. (Must be null-terminated for strto*).
516 SmallString<24> S;
517 S.push_back(First);
518 while (isNumber(peek()))
519 S.push_back(next());
520 char *End;
521 // Try first to parse as integer, and if so preserve full 64 bits.
522 // We check for errno for out of bounds errors and for End == S.end()
523 // to make sure that the numeric string is not malformed.
524 errno = 0;
525 int64_t I = std::strtoll(S.c_str(), &End, 10);
526 if (End == S.end() && errno != ERANGE) {
527 Out = int64_t(I);
528 return true;
529 }
530 // strtroull has a special handling for negative numbers, but in this
531 // case we don't want to do that because negative numbers were already
532 // handled in the previous block.
533 if (First != '-') {
534 errno = 0;
535 uint64_t UI = std::strtoull(S.c_str(), &End, 10);
536 if (End == S.end() && errno != ERANGE) {
537 Out = UI;
538 return true;
539 }
540 }
541 // If it's not an integer
542 Out = std::strtod(S.c_str(), &End);
543 return End == S.end() || parseError("Invalid JSON value (number?)");
544}

546bool Parser::parseString(std::string &Out) {
547 // leading quote was already consumed.
548 for (char C = next(); C != '"'; C = next()) {
549 if (LLVM_UNLIKELY(P == End))
550 return parseError("Unterminated string");
551 if (LLVM_UNLIKELY((C & 0x1f) == C))
552 return parseError("Control character in string");
553 if (LLVM_LIKELY(C != '\\')) {
554 Out.push_back(C);
555 continue;
556 }
557 // Handle escape sequence.
558 switch (C = next()) {
559 case '"':
560 case '\\':
561 case '/':
562 Out.push_back(C);
563 break;
564 case 'b':
565 Out.push_back('\b');
566 break;
567 case 'f':
568 Out.push_back('\f');
569 break;
570 case 'n':
571 Out.push_back('\n');
572 break;
573 case 'r':
574 Out.push_back('\r');
575 break;
576 case 't':
577 Out.push_back('\t');
578 break;
579 case 'u':
580 if (!parseUnicode(Out))
581 return false;
582 break;
583 default:
584 return parseError("Invalid escape sequence");
585 }
586 }
587 return true;
588}

590static void encodeUtf8(uint32_t Rune, std::string &Out) {
591 if (Rune < 0x80) {
592 Out.push_back(Rune & 0x7F);
593 } else if (Rune < 0x800) {
594 uint8_t FirstByte = 0xC0 | ((Rune & 0x7C0) >> 6);
595 uint8_t SecondByte = 0x80 | (Rune & 0x3F);
596 Out.push_back(FirstByte);
597 Out.push_back(SecondByte);
598 } else if (Rune < 0x10000) {
599 uint8_t FirstByte = 0xE0 | ((Rune & 0xF000) >> 12);
600 uint8_t SecondByte = 0x80 | ((Rune & 0xFC0) >> 6);
601 uint8_t ThirdByte = 0x80 | (Rune & 0x3F);
602 Out.push_back(FirstByte);
603 Out.push_back(SecondByte);
604 Out.push_back(ThirdByte);
605 } else if (Rune < 0x110000) {
606 uint8_t FirstByte = 0xF0 | ((Rune & 0x1F0000) >> 18);
607 uint8_t SecondByte = 0x80 | ((Rune & 0x3F000) >> 12);
608 uint8_t ThirdByte = 0x80 | ((Rune & 0xFC0) >> 6);
609 uint8_t FourthByte = 0x80 | (Rune & 0x3F);
610 Out.push_back(FirstByte);
611 Out.push_back(SecondByte);
612 Out.push_back(ThirdByte);
613 Out.push_back(FourthByte);
614 } else {
615 llvm_unreachable("Invalid codepoint");
616 }
617}

619// Parse a UTF-16 \uNNNN escape sequence. "\u" has already been consumed.
620// May parse several sequential escapes to ensure proper surrogate handling.
621// We do not use ConvertUTF.h, it can't accept and replace unpaired surrogates.
622// These are invalid Unicode but valid JSON (RFC 8259, section 8.2).
623bool Parser::parseUnicode(std::string &Out) {
624 // Invalid UTF is not a JSON error (RFC 8529§8.2). It gets replaced by U+FFFD.
625 auto Invalid = [&] { Out.append(/* UTF-8 */ {'\xef', '\xbf', '\xbd'}); };
626 // Decodes 4 hex digits from the stream into Out, returns false on error.
627 auto Parse4Hex = [this](uint16_t &Out) -> bool {
628 Out = 0;
629 char Bytes[] = {next(), next(), next(), next()};
630 for (unsigned char C : Bytes) {
631 if (!std::isxdigit(C))
632 return parseError("Invalid \\u escape sequence");
633 Out <<= 4;
634 Out |= (C > '9') ? (C & ~0x20) - 'A' + 10 : (C - '0');
635 }
636 return true;
637 };
638 uint16_t First; // UTF-16 code unit from the first \u escape.
639 if (!Parse4Hex(First))
640 return false;
641
642 // We loop to allow proper surrogate-pair error handling.
643 while (true) {
644 // Case 1: the UTF-16 code unit is already a codepoint in the BMP.
645 if (LLVM_LIKELY(First < 0xD800 || First >= 0xE000)) {
646 encodeUtf8(First, Out);
647 return true;
648 }
649
650 // Case 2: it's an (unpaired) trailing surrogate.
651 if (LLVM_UNLIKELY(First >= 0xDC00)) {
652 Invalid();
653 return true;
654 }
655
656 // Case 3: it's a leading surrogate. We expect a trailing one next.
657 // Case 3a: there's no trailing \u escape. Don't advance in the stream.
658 if (LLVM_UNLIKELY(P + 2 > End || *P != '\\' || *(P + 1) != 'u')) {
659 Invalid(); // Leading surrogate was unpaired.
660 return true;
661 }
662 P += 2;
663 uint16_t Second;
664 if (!Parse4Hex(Second))
665 return false;
666 // Case 3b: there was another \u escape, but it wasn't a trailing surrogate.
667 if (LLVM_UNLIKELY(Second < 0xDC00 || Second >= 0xE000)) {
668 Invalid(); // Leading surrogate was unpaired.
669 First = Second; // Second escape still needs to be processed.
670 continue;
671 }
672 // Case 3c: a valid surrogate pair encoding an astral codepoint.
673 encodeUtf8(0x10000 | ((First - 0xD800) << 10) | (Second - 0xDC00), Out);
674 return true;
675 }
676}

678bool Parser::parseError(const char *Msg) {
679 int Line = 1;
680 const char *StartOfLine = Start;
681 for (const char *X = Start; X < P; ++X) {
682 if (*X == 0x0A) {
683 ++Line;
684 StartOfLine = X + 1;
685 }
686 }
687 Err.emplace(
688 std::make_unique<ParseError>(Msg, Line, P - StartOfLine, P - Start));
689 return false;
690}

704bool isUTF8(llvm::StringRef S, size_t *ErrOffset) {
705 // Fast-path for ASCII, which is valid UTF-8.
706 if (LLVM_LIKELY(isASCII(S)))
707 return true;
708
709 const UTF8 *Data = reinterpret_cast<const UTF8 *>(S.data()), *Rest = Data;
710 if (LLVM_LIKELY(isLegalUTF8String(&Rest, Data + S.size())))
711 return true;
712
713 if (ErrOffset)
714 *ErrOffset = Rest - Data;
715 return false;
716}

718std::string fixUTF8(llvm::StringRef S) {
719 // This isn't particularly efficient, but is only for error-recovery.
720 std::vector<UTF32> Codepoints(S.size()); // 1 codepoint per byte suffices.
721 const UTF8 *In8 = reinterpret_cast<const UTF8 *>(S.data());
722 UTF32 *Out32 = Codepoints.data();
723 ConvertUTF8toUTF32(&In8, In8 + S.size(), &Out32, Out32 + Codepoints.size(),
724 lenientConversion);
725 Codepoints.resize(Out32 - Codepoints.data());
726 std::string Res(4 * Codepoints.size(), 0); // 4 bytes per codepoint suffice
727 const UTF32 *In32 = Codepoints.data();
728 UTF8 *Out8 = reinterpret_cast<UTF8 *>(&Res[0]);
729 ConvertUTF32toUTF8(&In32, In32 + Codepoints.size(), &Out8, Out8 + Res.size(),

735static void quote(llvm::raw_ostream &OS, llvm::StringRef S) {
736 OS << '\"';
737 for (unsigned char C : S) {
738 if (C == 0x22 || C == 0x5C)
739 OS << '\\';
740 if (C >= 0x20) {
741 OS << C;
742 continue;
743 }
744 OS << '\\';
745 switch (C) {
746 // A few characters are common enough to make short escapes worthwhile.
747 case '\t':
748 OS << 't';
749 break;
750 case '\n':
751 OS << 'n';
752 break;
753 case '\r':
754 OS << 'r';
755 break;
756 default:
757 OS << 'u';
758 llvm::write_hex(OS, C, llvm::HexPrintStyle::Lower, 4);
759 break;
760 }
761 }
762 OS << '\"';
763}

765void llvm::json::OStream::value(const Value &V) {
766 switch (V.kind()) {
767 case Value::Null:
768 valueBegin();
769 OS << "null";
770 return;
771 case Value::Boolean:
772 valueBegin();
773 OS << (*V.getAsBoolean() ? "true" : "false");
774 return;
775 case Value::Number:
776 valueBegin();
777 if (V.Type == Value::T_Integer)
778 OS << *V.getAsInteger();
779 else if (V.Type == Value::T_UINT64)
780 OS << *V.getAsUINT64();
781 else
782 OS << format("%.*g", std::numeric_limits<double>::max_digits10,
783 *V.getAsNumber());
784 return;
785 case Value::String:
786 valueBegin();
787 quote(OS, *V.getAsString());
788 return;
789 case Value::Array:
790 return array([&] {
791 for (const Value &E : *V.getAsArray())
792 value(E);
793 });
794 case Value::Object:
795 return object([&] {
796 for (const Object::value_type *E : sortedElements(*V.getAsObject()))
797 attribute(E->first, E->second);
798 });
799 }
800}

819void OStream::flushComment() {
820 if (PendingComment.empty())
821 return;
822 OS << (IndentSize ? "/* " : "/*");
823 // Be sure not to accidentally emit "*/". Transform to "* /".
824 while (!PendingComment.empty()) {
825 auto Pos = PendingComment.find("*/");
826 if (Pos == StringRef::npos) {
827 OS << PendingComment;
828 PendingComment = "";
829 } else {
830 OS << PendingComment.take_front(Pos) << "* /";
831 PendingComment = PendingComment.drop_front(Pos + 2);
832 }
833 }
834 OS << (IndentSize ? " */" : "*/");
835 // Comments are on their own line unless attached to an attribute value.
836 if (Stack.size() > 1 && Stack.back().Ctx == Singleton) {
837 if (IndentSize)
838 OS << ' ';
839 } else {
840 newline();
841 }
842}

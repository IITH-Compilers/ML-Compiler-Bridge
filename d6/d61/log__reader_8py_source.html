39@dataclasses.dataclass(frozen=True)
40class TensorSpec:
41 name: str
42 port: int
43 shape: List[int]
44 element_type: type
45
46 @staticmethod
47 def from_dict(d: dict):
48 name = d["name"]
49 port = d["port"]
50 shape = [int(e) for e in d["shape"]]
51 element_type_str = d["type"]
52 if element_type_str not in _element_types:
53 raise ValueError(f"uknown type: {element_type_str}")
54 return TensorSpec(
55 name=name,
56 port=port,
57 shape=shape,
58 element_type=_element_types[element_type_str],
59 )

95def read_header(f: io.BufferedReader):
96 line = f.readline()
97 header = json.loads(line)
98 tensor_specs = [TensorSpec.from_dict(ts) for ts in header["features"]]
99 score_spec = TensorSpec.from_dict(header["score"]) if "score" in header else None
100 advice_spec = TensorSpec.from_dict(header["advice"]) if "advice" in header else None
101 return tensor_specs, score_spec, advice_spec

118def read_stream(fname: str):
119 with io.BufferedReader(io.FileIO(fname, "rb")) as f:
120 tensor_specs, score_spec, _ = read_header(f)
121 context = None
